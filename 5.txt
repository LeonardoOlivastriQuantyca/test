SELECT NomeS, AnnoN
FROM studenti
WHERE CorsoLaurea = 'SBC'
ORDER BY s.NomeS

--- 2 ---

SELECT DISTINCT d.NomeD, d.Dipartimento
FROM docenti d
JOIN corsi c
ON d.CodDocente = c.CodDocente
WHERE c.NomeCorso IN ('Basi di Dati e Sistemi Informativi', 'Informatica Generale')

--- 3 ---

SELECT s.Matricola, s.NomeS
FROM studenti s
JOIN corsodilaurea l
ON s.CorsoLaurea=l.CorsoLaurea
JOIN frequenta f
ON s.Matricola=f.Matricola
JOIN corsi c
ON f.CodCorso=c.CodCorso
JOIN docenti d
ON d.CodDocente=c.CodDocente
WHERE l.TipoLaurea='L' AND
d.NomeD LIKE '%Felice%'

--- 4 ---

SELECT l.CorsoLaurea, l.TipoLaurea, AVG(2024-s.AnnoN) AS EtàMedia
FROM corsodilaurea l
JOIN studenti s
ON l.CorsoLaurea=s.CorsoLaurea
GROUP BY l.CorsoLaurea

--- 5 ---

SELECT c.CodCorso, COUNT(*) AS NumStudenti
FROM corsi c
JOIN docenti d
ON c.CodDocente=d.CodDocente
JOIN frequenta f
ON c.CodCorso=f.CodCorso
WHERE d.NomeD LIKE '%Leoni%'
GROUP BY c.CodCorso

--- 6 ---

SELECT c.CodCorso
FROM corsi c
JOIN docenti d
ON c.CodDocente=d.CodDocente
JOIN frequenta f
ON c.CodCorso=f.CodCorso
WHERE d.Dipartimento = 'Informatica'
GROUP BY c.CodCorso
HAVING COUNT(*) > 5

--- 7 ---

SELECT s.Matricola, COUNT(*) AS NumCorsi
FROM studenti s
JOIN corsodilaurea l
ON s.CorsoLaurea=l.CorsoLaurea
JOIN frequenta f
ON s.Matricola=f.Matricola
WHERE l.Facoltà = 'Lettere e Filosofia'
GROUP BY s.Matricola

--- 8 ---

SELECT s.Matricola, s.NomeS
FROM studenti s
LEFT OUTER JOIN frequenta f
ON s.Matricola=f.Matricola

--- 9 ---

SELECT d.CodDocente, d.NomeD
FROM docenti d
JOIN corsi c
ON d.CodDocente=c.CodDocente
LEFT OUTER JOIN frequenta
ON c.CodCorso = f.CodCorso

--- 10 ---

SELECT s.Matricola, s.NomeS
FROM studenti s
JOIN frequenta f
ON s.Matricola=f.Matricola
JOIN corsi c
ON f.CodCorso=c.CodCorso
JOIN docenti d
ON c.CodDocente=d.CodDocente
WHERE d.CodDocente IN (
	SELECT d.CodDocente
	FROM docenti d
	WHERE d.Dipartimento = 'Storia'
)

--- 11 ---

SELECT c.CodCorso
FROM corsi c
JOIN frequenta f
ON c.CodCorso=f.CodCorso
JOIN studenti s
ON f.Matricola=s.Matricola
JOIN corsidilaurea l
ON s.CorsoLaurea=l.CorsoLaurea
WHERE EXISTS(
	SELECT *
	FROM etc
	WHERE l.TipoLaurea = 'L' AND
	s.CorsoLaurea = 'SBC'
)

--- 12 ---

SELECT d.NomeD, d.CodDocente
FROM docenti d
JOIN corsi c
ON d.CodDocente=c.CodDocente
JOIN frequenta f
ON c.CodCorso=f.CodCorso
GROUP BY f.CodCorso, d.NomeD, d.CodDocente
HAVING COUNT(*) > 5

--- 13 ---

SELECT f.CodCorso
FROM frequenta f
GROUP BY f.CodCorso
HAVING COUNT(*) = (
	SELECT COUNT(*)
	FROM studenti s
	WHERE s.CorsoLaurea = 'SBC'
)
